kosaraju算法

首先放出无向图的连通性判断代码。

基本思路是从任意一个顶点开始DFS，每次DFS标记所有包含这个结点的连通分量的所有结点，最终的连通分量个数就是DFS总次数。 

#include <stdio.h>
#include <string.h>
#define MAX 100
int p[MAX][MAX];
int vis[MAX];
int v, e;

void DFS(int s)
{
    for(int u=0; u<v; u++)
    {
        if(!vis[u] & p[s][u])
        {
            vis[u] = 1;
            DFS(u);
        }
    }
    return;
}

int main(void)
{
    memset(p,0,sizeof(p));
    memset(vis,0,sizeof(vis));
    printf("请输入节点数和边的数目：\n");
    scanf("%d%d",&v,&e);
    int s, t;
    printf("请输入边：\n");
    for(int i=0; i<e; i++)
    {
        scanf("%d%d",&s,&t);
        p[s][t] = 1;
        p[t][s] = 1;
    }
    int cnt = 0;
    for(int i=0; i<v; i++)
    {
        if(!vis[i])
        {
            DFS(i);
            cnt++;
        }
    }
    if(cnt==1) printf("是连通图。\n");
    else printf("不是连通图，连通分量为%d。\n", cnt);
    return 0;
}

再思考如何解决有向图的强连通分量数目问题。显然不能够直接套用无向图的方法解决，因为这样有可能出现明明有多个强连通分量却因为遍历顺序随机而导致一次DFS就能够标记所有结点。

因为在上图中有两个强连通分量A和B，B是被A指向的分量。这就导致，当我们首先从A中结点开始遍历的时候出现了问题。为了解决这个问题，我们需要让被指向的强连通分量优先被搜索到，即B中至少一个顶点排在A中所有顶点的前面进行DFS。

那么如何得到这个特定的顺序呢？这就需要对原图G进行一遍DFS操作，找到图中结点的逆后序。 

/*  
 * 如果当前顶点未访问，先遍历完与当前顶点相连的且未被访问的所有其它顶点，
 * 然后将当前顶点加入栈中，最后栈中从栈顶到栈底的顺序就是我们需要的顶点顺序。
 * 实际上这个逆后序就是第一次DFS后各节点的finishing time的降序排列。
 */
void dfs_1(int u)
{
    vis[u] = 1;
    for(int i=0; i<v; i++)
        if(!vis[i] && map[u][i])
            dfs_1(i);
    s.push(u);
}

有了这个逆后序，我们只需要再在原图G的转置上进行和无向图一样的DFS操作就可以得到有向图的强连通分量啦。(转置图的强连通分量和原图是完全一样的)

#include <iostream>
#include <cstdio>
#include <cstring>
#include <stack>
using namespace std;

#define MAX 100
int v,e;
int map[MAX][MAX], nmap[MAX][MAX], vis[MAX];
//邻接矩阵 邻接矩阵的转置 标记
stack <int> s;//栈 用于记录逆后序

void dfs_1(int u)//用于求逆后序
{
    vis[u] = 1;
    for(int i=0; i<v; i++)
        if(!vis[i] && map[u][i])
            dfs_1(i);
    s.push(u);
}

void dfs_2(int u)//遍历 map的转置
{
    vis[u] = 1;
    for(int i=0; i<v; i++)
        if(!vis[i] && nmap[u][i])
            dfs_2(i);
}

int kosaraju()
{
    while(!s.empty())
        s.pop();
    memset(vis,0,sizeof(vis));//初始化

    for(int i=0; i<v; i++)
        if(!vis[i])
            dfs_1(i);//入栈
    
    int ans = 0;
    memset(vis, 0, sizeof(vis));

    while(!s.empty())//遍历图内每一个点
    {
        int u = s.top();//每次取栈顶的顶点
        s.pop();
        //int u = t[i++];
        if(!vis[u])//如果该顶点在之前没有被访问过
        {
            ans++;//强连通分量++
            dfs_2(u);//从该点开始DFS 标记和该顶点同属一个强连通分量的顶点
        }
    }
    return ans;//返回答案
}

int main()
{
    memset(map,0,sizeof(map));
    memset(nmap,0,sizeof(nmap));

    printf("请输入节点数和边的数目：\n");
    scanf("%d%d",&v,&e);
    int s, t;
    printf("请输入边：\n");
    for(int i=0; i<e; i++)
    {
        scanf("%d%d",&s,&t);
        map[s][t] = 1;
        nmap[t][s] = 1;
    }
    int cnt = kosaraju();
    if(cnt == 1) printf("是强连通图。\n");
    else printf("不是强连通图，强连通分量为%d。\n",cnt);
    return 0;
}
