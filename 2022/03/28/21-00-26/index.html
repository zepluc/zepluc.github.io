栈和队列学习笔记

栈和队列

限定插入和删除只能在表的端点进行的线性表。

栈

限定在表尾作插入、删除操作的线性表。后进先出。

  • 数据对象： \(D\) = {\(a_i|a_i\in ElemSet,i=1,2\cdots ,n,n\ge 0\)}

  • 数据关系： \(R_1\)={\(<a_i,a_{i-1}>\)|\(a_i,a_{i-1}\in D,i=1,2,\cdots n\)}

  • 基本操作

    • Initstack(&s);
    • Push(&s,e);
    • Pop(&s,&e);
    • GetTop(s,&e);
    • StackEmpty(s);

顺序栈

  • 定义
/*静态分配*/
typedef struct
{
    SElemType elem[maxlen];
    int top;
}SqStack;
/*动态分配*/
#define STACK_INIT_SIZE 100
#define STACKINCREMENT 10
typedef struct
{
    SElemType *base;    //指向栈空间基地址
    SElemType *top;     //指向栈顶元素的后一单元
    int stacksize;      //已分配的存储空间大小
}SqStack;
  • 初始化栈（动态分配）
Status InitStack(SqStack &S)
{
    S.base = (SElemType *)malloc(STACK_INIT_SIZE*sizeof(SElemType));
    if(!S.base) exit(OVERFLOW);
    S.top = S.base;
    S.size = STACK_INIT_SIZE;
    return OK;
}
  • 取栈顶元素
Status GetTop(SqStack &S)
{
    if(S.top==S.base) return ERROR; //空栈
    e  = *(S.top-1);
    return OK;
}
  • 进栈
Status Push(SqStack &S,SElemType e)
{
    if(S.top - S.base >= S.stacksize)
    {
        SElemType* newbase;
        newbase = (SElemType*) realloc(S.base, (S.stacksize+STACKINCREMENT)*sizeof(SElemType));
        if(!newbase) exit(OVERFLOW);
        S.base = newbase;
        S.top = S.base + S.stacksize;
        S.stacksize += STACKINCREMENT;
    }
    *S.top++ = e;
    return OK;
}
  • 出栈
Status Pop(SqStack &S,SElemType &e)
{
    if(S.top==S.base) return ERROR;
    e = *--S.top;
    return OK;
}

链式栈

不带表头节点的单链表

  • 定义
struct node
{
    SElemType data;
    struct node *next;
}*top = NULL;   //初始化 置top为空栈
  • 进栈
struct node* push_link(struct node *top, SElemType e)
{
    struct node p;
    p = (struct node*) malloc(sizeof(struct nodes));
    p -> data = e; //装入元素e
    p -> next = top; //入栈
    top = p; //变头节点
    return top;
}
  • 出栈
struct node* pop(struct node *p, SElemType *e)
{
    struct node *p;
    if(top==NULL) return NULL;//空栈 返回NULL
    p = top;
    (*e) = p->data;
    top = top->next;
    free(p);
    return top;
}

队列

限定在表头删除，在表尾插入的线性表。先进先出。

  • 数据对象： \(D\) = {\(a_i|a_i\in ElemSet,i=1,2\cdots ,n,n\ge 0\)}

  • 数据关系： \(R_1\)={\(<a_i,a_{i-1}>\)|\(a_i,a_{i-1}\in D,i=1,2,\cdots n\)}

  • 基本操作

    • InitQueue(&Q)
    • DestroyQueue(&Q)
    • QueueEmpty(Q)
    • QueueLength(Q)
    • GetHead(Q,&e)
    • EnQueue(&Q,e)
    • DeQueue(&Q,&e)
    • ClearQueue(&Q)
    • QueueTraverse(Q,visit())

链式队列

带表头节点的单链表,当Q.front==Q.rear时为空表。

  • 定义
/*存放元素的结点类型*/
typedef struct Qnode
{
    QElemType data;
    struct Qnode *next;
}Qnode,*QueuePtr;//节点类型和指针类型
/*由头、尾指针组成的队列类型*/
typedef struct
{
    Qnode *front;
    Qnode *rear;
}LinkQueue;//链式队列类型
  • 生成空队列
Status InitQueue(LinkQueue &Q)
{
    Q.front = Q.rear = (QueuePtr*) malloc(sizeof(Qnode));
    if(!Q.front) exit(OVERFLOW);
    Q.front -> next = NULL;
    return OK;
}
  • 入队
Status EnQueue(LinkQueue &Q, QElemType e)
{
    p = (QueuePtr) malloc(sizeof(Qnode));
    if(!p) exit(OVERFLOW);
    p -> data = e;
    p -> next = NULL;
    Q.rear -> next = p;
    Q.rear = p;
    return OK;
}
  • 出队
Status DeQueue(LinkQueue &Q, QElemType &e)
{
    if(Q.front == Q.rear) return ERROR;
    QueuePtr p = Q.front->next;
    e = p->data;
    Q.front -> next = p ->next;
    if(Q.rear == p) //如果原队列只有一个结点的时候需要改变rear指针！
        Q.rear = Q.front;
    free(p);
    return OK;
}

循环队列

解决假溢出，形成头尾相接的顺序存储结构。 问题在于，无论队空还是队满都会出现rear == front的情况。为了解决这个问题，可以保留一个元素空间，即(rear+1)%MAXQSIZE==front的时候就判队满。

  • 定义
#define MAXQSIZE 100
typedef struct
{
    QElemType *base;
    int front;
    int rear;
}SqQueue, Q;
  • 构造空队列
Status InitQueue (SqQueue &Q)
{
    Q.base = (QElemType*) malloc(MAXQSIZE*sizeof(QElemType));
    if(!Q.base) exit(OVERFLOW);
    Q.front = Q.rear;
    return OK;
}
  • 求队列长度
int QueueLength(SqQueue Q)
{
    return (Q.rear - Q.front + MAXQSIZE) % MAXQSIZE;
}
  • 插入队尾元素
Status EnQueue (SqQueue &Q, QElemType e)
{
    if((Q.rear+1)%MAXQSIZE == Q.front) return ERROR;//队列满
    Q.base[Q.rear] = e;
    Q.rear = (Q.rear+1) % MAXQSIZE;//尾指针循环后移
    return OK;
}
  • 删除队头元素
Status DeQueue (SqQueue &Q, QElemType e)
{
    if(Q.front == Q.rear) return ERROR;
    e = Q.base[Q.front];
    Q.front = (Q.front+1) % MAXQSIZE;//头指针循环后移
    return OK;
}
